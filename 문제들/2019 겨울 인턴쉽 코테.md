    ```cpp
    #include <vector>
    #include <set>
    #include <unordered_set>

    using namespace std;

    vector<long long> solution(long long k, vector<long long> room_number) {
        vector<long long> answer;
        unordered_set<long long> s;
        
        for(int i=0; i<room_number.size(); i++){
            if(s.find(room_number[i]) == s.end()) {
                s.insert(room_number[i]);
                answer.push_back(room_number[i]);
            }else{
                int x = room_number[i];
                while(s.find(x) != s.end()){
                    x++;
                }
                s.insert(x);
                answer.push_back(x);
            }
        }
        
        return answer;
    }
    ```

## 크레인 인형뽑기 게임

[프로그래머스](https://programmers.co.kr/learn/courses/30/lessons/64061)

- 코드 좀 더러운듯 그치만 정답

    ```cpp
    #include <string>
    #include <vector>

    using namespace std;

    int solution(vector<vector<int>> board, vector<int> moves) {
        int answer = 0;
        int N = board.size();
        vector<int> basket;
        for(int i=0; i<moves.size(); i++){
            int col = moves[i] - 1;
            for(int row=0; row<N; row++){
                if(board[row][col] == 0) continue;
                if(!basket.empty()){
                    if(basket.back() == board[row][col]){
                        basket.pop_back();
                        board[row][col] = 0;
                        answer += 2;
                    }else {
                        basket.push_back(board[row][col]);
                        board[row][col] = 0;
                    }
                }else if(basket.empty()){
                    basket.push_back(board[row][col]);
                    board[row][col] = 0;
                }
                break;
            }
        }
        return answer;
    }
    ```

## 튜플

[프로그래머스](https://programmers.co.kr/learn/courses/30/lessons/64065)

- 시간 좀 걸렸지만 그래도 통과

    ```cpp
    #include <string>
    #include <vector>
    #include <algorithm>
    #include <unordered_set>

    using namespace std;

    vector<vector<int> > result;

    void splitS(string s) {
    	vector<int> tmp;
    	string val = "";
    	for (int i = 0; i < s.size(); i++) {
    		if (s[i] >= '0' && s[i] <= '9') {
    			val += s[i];
    		}
    		else if (s[i] == ',' && !val.empty()) {
    			tmp.push_back(stoi(val));
    			val = "";
    		}
    		else if (s[i] == '}') {
    			tmp.push_back(stoi(val));
    			val = "";
    			result.push_back(tmp);
    			tmp.clear();
    		}
    	}
    }

    bool comp(vector<int> a, vector<int> b){
        return a.size() < b.size();
    }

    vector<int> solution(string s) {
        vector<int> answer;
        unordered_set<int> se;
        
        string origin = s.substr(1,s.size()-2);
    	splitS(origin);
        sort(result.begin(), result.end(), comp);
        for (int i = 0; i < result.size(); i++) {
    		for (int j = 0; j < result[i].size(); j++) {
    			if(answer.empty()) {
                    answer.push_back(result[i][j]);
                    se.insert(result[i][j]);
                }
                else{
                    if(se.find(result[i][j]) == se.end()){
                       answer.push_back(result[i][j]);
                        se.insert(result[i][j]); 
                    }
                }
    		}
    	}
        
        return answer;
    }
    ```
